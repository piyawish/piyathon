คะแนนรวม = 0

และคิดราคาดังนี้: 0-50 หน่วย คิดหน่วยละ 3 บาท, 51-100 หน่วย คิดหน่วยละ 4 บาท, มากกว่า 100 หน่วย คิดหน่วยละ 5 บาท

หน่วย = จำนวนเต็ม(รับข้อมูล("ป้อนจำนวนหน่วยไฟฟ้า: "))
ค่าไฟ = 0

ถ้า หน่วย <= 50:
    ค่าไฟ = หน่วย * 3
อื่นถ้า หน่วย <= 100:
    ค่าไฟ = (50 * 3) + ((หน่วย - 50) * 4)
อื่น:
    ค่าไฟ = (50 * 3) + (50 * 4) + ((หน่วย - 100) * 5)

พิมพ์(f"ค่าไฟฟ้า = {ค่าไฟ} บาท")
